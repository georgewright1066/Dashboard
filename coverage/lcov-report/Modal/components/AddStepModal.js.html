import { connect } from 'react-redux';
import React from 'react';
import DropDown from '../../common/components/PrettyDropdown';
import Button from '../../common/components/Button';
import classNames from 'classnames';
import URL_REGEX from '../../CellStepManagement/duck/constants';
import External from '../../CellStepManagement/components/External';
import CachePage from '../../CellStepManagement/components/CachePage';
import StimTable from '../../CellStepManagement/components/StimTable';
import { CellStepManagementOperations } from '../../CellStepManagement/duck/index';
 
 
class Form extends React.Component {
 
  state = {
    value: 'Please Select',
    stepType: '',
    error: false,
    step: 0,
    param: '',
    secondaryValue: 'Please Select',
    url: ''
  }
 
  onChange = (e) => {
    const url = e.target.value
    this.setState({ url })
  }
 
  onParamChange = (e) => {
    const param = e.target.value
    this.setState({ param })
  }
 
  handleClick = (id, value, studyId, cacheId) => {
 
    const { stepType, param, url, secondaryValue } = this.state;
 
    switch (stepType) {
      case ('Validation'):
        this.onHandleClickComplete(id, value, studyId, stepType, null)
        break;
      case ('Embedded'):
        if (!this.validateUrl(url)) {
          this.setState({ error: true })
        } else {
          value = url;
          this.onHandleClickComplete(id, value, studyId, stepType, param);
        }
        break;
      case ('External'):
        if (!this.validateUrl(url)) {
          this.setState({ error: true })
        } else {
          value = url
          this.onHandleClickComplete(id, value, studyId, stepType, param);
        }
        break;
      case 'Instruction':
        value = secondaryValue;
        this.onHandleClickComplete(id, value, studyId, stepType, null)
        break;
      case 'Cache Page':
        value = `cache:${id}`
        this.onHandleClickComplete(id, value, studyId, stepType, cacheId)
        break;
      case "Stim":
        value = `stim:${id}`
        this.onHandleClickComplete(id, value, studyId, stepType, cacheId)
        break
      default:
        this.setState({ error: true })
    }
  }
 
  onHandleClickComplete(id, value, studyId, stepType, cacheId) {
    this.props.addTask(id, value, studyId, stepType, cacheId)
    this.resetForm()
  }
 
 
  onSelectItem = (stepType) => {
    this.props.resetStateToLoading();
    const studyId = localStorage.getItem('studyId');
    this.setState({ stepType: stepType, value: stepType })
    this.props.onSelectItem(stepType, studyId)
    if (stepType === 'External' || 'Instuction') {
      this.setState({ error: false })
    }
 
  }
 
  validateUrl(url) {
    return URL_REGEX.test(url);
  }
 
  resetForm() {
    this.props.closeModal()
    this.setState({ stepType: '', value: 'Please Select', url: '', error: false })
 
  }
 
  render() {
    const { modalOpen, id, dataFromPrimarySelection, studyId, loading, options } = this.props;
    const { value, stepType, error, url } = this.state;
    const showSubmitButton = (stepType === 'Instruction') || (stepType === 'External') || (stepType === 'Embedded' || (stepType === 'Validation')) ? true : false;
    return (
      <form className={classNames('csm-modal-form', { 'active': modalOpen })}>
        <Button handleClick={() => this.resetForm()} buttonClass="filter__exit-button" />
        <div className="csm-modal-form__content-container">
          <h2>Step Types {id}</h2>
          <DropDown title={this.state.value} onSelectItem={(item) => this.onSelectItem(item, studyId)} options={options} />
          {
            (() => {
              switch (stepType) {
                case ('External'):
                  return (<External value={url} onChange={this.onChange} onParamChange={this.onParamChange} />);
                case ('Embedded'):
                  return (<External value={url} onChange={this.onChange} onParamChange={this.onParamChange} />);
                case 'Instruction':
                  return (<DropDown largeContent={true} title={this.state.secondaryValue} onSelectItem={(item) => this.setState({ secondaryValue: item })} options={dataFromPrimarySelection} />);
                case 'Cache Page':
                  return (<CachePage onClick={(cacheId) => this.handleClick(id, stepType, studyId, cacheId)} loading={loading} cachePage={dataFromPrimarySelection} />)
                case "Stim":
                  return (<StimTable onClick={(cacheId) => this.handleClick(id, stepType, studyId, cacheId)} loading={loading} data={dataFromPrimarySelection} />)
                default:
                  return null
              }
            }
            )()
          }
          <div className={classNames('csm__error-message', { 'active': error })}>Please enter a valid URL</div>
          {showSubmitButton ?
            <Button handleClick={() => this.handleClick(id, value, studyId)} text="Submit" buttonClass="button-primary show-study__button" /> :
            null
          }
 
        </div>
      </form>
    );
  }
}
 
const mapDispatchToProps = (dispatch, ownProps) => {
  return {
    closeModal: () => (dispatch(CellStepManagementOperations.closeModal())),
    resetStateToLoading: () => dispatch(CellStepManagementOperations.resetStateToLoading()),
    onSelectItem: (item, id) => (dispatch(CellStepManagementOperations.onSelectItem(item, id))),
    addTask: (id, value, studyId, item, cacheId) => (dispatch(CellStepManagementOperations.addTask(id, value, studyId, item, cacheId))),
  };
};
 
const mapStateToProps = (state, ownProps) => {
  return {
    modalOpen: state.cellStepManagement.modalOpen,
    id: state.cellStepManagement.id,
    loading: state.cellStepManagement.loading,
    dataFromPrimarySelection: state.cellStepManagement.dataFromPrimarySelection,
    options: state.cellStepManagement.initialDropDown,
 
  };
};
 
export default connect(mapStateToProps, mapDispatchToProps)(Form);
